الدرس 2: التعامل مع DOM

المدة: 25 دقيقة | المستوى: مهم جداً

1. يعني إيه DOM؟

DOM = Document Object Model
ده تمثيل لصفحة HTML بتاعتك على شكل "شجرة".
JS بيقدر يمسك أي عنصر في الشجرة دي ويغيره وانت فاتح الموقع.

بالمصري: HTML ثابت ميت، DOM بيخليه حي وبيتحرك 😂

2. ازاي تمسك عنصر من الصفحة؟

فيه 3 طرق أساسية. احفظهم زي اسمك:

// 1. بالـ ID - الأسرع والأشهر
document.getElementById("myTitle")

// 2. بالـ Class
document.querySelector(".my-class")

// 3. بالتاج نفسه
document.querySelector("h1")
querySelector = الأسهل. بيجيب أول عنصر يلاقيه
querySelectorAll = بيجيب كل العناصر اللي بنفس الاسم

3. غير النص بتاع عنصر

let title = document.getElementById("myTitle");
title.textContent = "النص الجديد"; // غير النص
title.innerHTML = "<span>نص + HTML</span>"; // غير بـ HTML

النص ده هيتغير لما تدوس

4. غير الستايل CSS بالـ JS

تقدر تغير لون، حجم، أي حاجة:

let box = document.getElementById("myBox");
box.style.background = "#00FF88"; // غير اللون
box.style.color = "#000"; // لون النص
box.style.fontSize = "30px"; // حجم الخط
البوكس ده لونه هيتغير

5. ضيف واحذف Classes

دي أقوى طريقة عشان تستخدم CSS جاهز:

// ضيف كلاس
element.classList.add("active");

// احذف كلاس
element.classList.remove("active");

// بدل بينهم
element.classList.toggle("dark-mode");

6. اعمل عنصر جديد من الصفر

// 1. اعمل العنصر
let newDiv = document.createElement("div");

// 2. اديله محتوى
newDiv.textContent = "أنا عنصر جديد 😂";

// 3. اديله كلاس
newDiv.classList.add("my-box");

// 4. حطه في الصفحة
document.body.appendChild(newDiv);

7. الـ Events = لما المستخدم يعمل حاجة

لما يدوس، يكتب، يحرك الماوس... كله Events:

let btn = document.getElementById("myBtn");

// الطريقة الأولى
btn.onclick = function() {
  alert("دوست على الزرار!");
}

// الطريقة الأحدث - الأفضل
btn.addEventListener("click", function() {
  alert("دوست على الزرار!");
});
