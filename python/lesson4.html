بايثون تقدر تقرأ وتكتب أي ملف على جهازك... txt, csv, json كله

1. قراءة ملف txt

لو عندك ملف اسمه data.txt فيه:

تقراه كده:

# الطريقة الصح - with بتقفل الملف لوحدها

    print(content)

2. كتابة ملف جديد

with open("output.txt", "w", encoding="utf-8") as file:
    file.write("أول سطر\n")
    file.write("تاني سطر\n")
    file.write("تالت سطر 🔥")

هيعمل ملف جديد اسمه output.txt

"w" = Write بيمسح القديم ويكتب جديد
"a" = Append بيضيف على القديم
"r" = Read للقراءة بس
encoding="utf-8" = عشان العربي يظهر صح ⚠️

4. التعامل مع JSON

JSON = أهم فورمات لتخزين البيانات

import json

# نكتب JSON
data = {
    "name": "جابر",
    "age": 20,
    "skills": ["Python", "JS", "HTML"]
}

with open("user.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=4)

# نقرأ JSON
with open("user.json", "r", encoding="utf-8") as f:
    user = json.load(f)
    print(user["name"]) # جابر
    print(user["skills"]) # ['Python', 'JS', 'HTML']

5. مشروع عملي: سجل مهام في ملف

import json
from datetime import datetime

def save_tasks(tasks):
    with open("tasks.json", "w", encoding="utf-8") as f:
        json.dump(tasks, f, ensure_ascii=False, indent=4)

def load_tasks():
    try:
        with open("tasks.json", "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return []

# نجرب
tasks = load_tasks()
tasks.append({
    "task": "اذاكر Python",
    "date": str(datetime.now()),
    "done": False
})
save_tasks(tasks)
print("تم حفظ المهمة ✅")
