اعمل ملف اسمه bot.py وحط فيه الكود ده:

from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes
import json
import os
from datetime import datetime

# التوكن بتاعك من BotFather
TOKEN = "حط_التوكن_بتاعك_هنا"

# ملف نحفظ فيه المستخدمين
USERS_FILE = "users.json"

def load_users():
    if os.path.exists(USERS_FILE):
        with open(USERS_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    return {}

def save_users(users):
    with open(USERS_FILE, "w", encoding="utf-8") as f:
        json.dump(users, f, ensure_ascii=False, indent=4)

# أمر /start
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user = update.effective_user
    users = load_users()

    # نسجل المستخدم
    user_id = str(user.id)
    if user_id not in users:
        users[user_id] = {
            "name": user.first_name,
            "username": user.username,
            "joined": str(datetime.now())
        }
        save_users(users)

    await update.message.reply_text(
        f"أهلاً {user.first_name} 🔥\n"
        f"أنا بوت جابر، بكتب بايثون 😂\n\n"
        f"الأوامر المتاحة:\n"
        f"/help - المساعدة\n"
        f"/stats - عدد المستخدمين\n"
        f"/calc 5+3 - آلة حاسبة"
    )

# أمر /help
async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "الأوامر:\n"
        "/start - نبدأ\n"
        "/stats - الإحصائيات\n"
        "/calc - آلة حاسبة: /calc 10*5\n"
        "ابعت أي رسالة وهرد عليك"
    )

# أمر /stats
async def stats(update: Update, context: ContextTypes.DEFAULT_TYPE):
    users = load_users()
    count = len(users)
    await update.message.reply_text(f"عدد المستخدمين: {count} 👥")

# أمر /calc
async def calc(update: Update, context: ContextTypes.DEFAULT_TYPE):
    try:
        expression = ' '.join(context.args)
        result = eval(expression) # ⚠️ خطر في الحقيقة
        await update.message.reply_text(f"النتيجة: {result}")
    except:
        await update.message.reply_text("اكتب العملية صح: /calc 5+3")

# الرد على أي رسالة عادية
async def echo(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.lower()

    if "ازيك" in text or "عامل ايه" in text:
        await update.message.reply_text("تمام يا معلم انت عامل ايه؟ 😂💚")
    elif "بايثون" in text:
        await update.message.reply_text("بايثون دي أحلى لغة 🐍🔥")
    elif "جابر" in text:
        await update.message.reply_text("جابر ده اللي عملني 😂👑")
    else:
        await update.message.reply_text(f"انت قولت: {update.message.text}")

# تشغيل البوت
def main():
    app = Application.builder().token(TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(CommandHandler("stats", stats))
    app.add_handler(CommandHandler("calc", calc))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, echo))

    print("البوت شغال... 🔥")
    app.run_polling()

if __name__ == "__main__":
    main()

def translate(text):
    url = f"https://api.mymemory.translated.net/get?q={text}&langpair=ar|en"
    res = requests.get(url).json()
    return res['responseData']['translatedText']
