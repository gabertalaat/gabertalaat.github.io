2. Function بـ Parameters

    print(f"اسمك {name} وعمرك {age} سنة")

greet("جابر", 20)
greet("أحمد", 25)
اسمك جابر وعمرك 20 سنة
اسمك أحمد وعمرك 25 سنة

3. Function بترجع قيمة - return

def add(x, y):
    result = x + y
    return result

sum = add(5, 3)
print(sum) # 8

# ممكن تستخدمها علطول
print(add(10, 20)) # 30

4. Default Parameters

def power(base, exp=2):
    return base ** exp

print(power(5)) # 25 ← exp افتراضياً 2
print(power(5, 3)) # 125 ← 5^3
** يعني أس: 5 ** 2 = 25
* يعني ضرب: 5 * 2 = 10

الجزء الثاني: OOP - البرمجة الكائنية 🏗️

OOP = بدل ما تكتب كود سايح، بتعمل "قوالب" Classes وتصنع منها Objects

1. أول Class ليك

class User:
    def __init__(self, name, age):
        self.name = name # خاصية
        self.age = age # خاصية

    def say_hello(self): # method
        print(f"أنا {self.name} وعندي {self.age} سنة")

# نصنع objects من الـ Class
user1 = User("جابر", 20)
user2 = User("سارة", 22)

user1.say_hello() # أنا جابر وعندي 20 سنة
user2.say_hello() # أنا سارة وعندي 22 سنة
__init__ = الكونستركتور، بيتنادى أول ما تصنع object جديد
self = بيشاور على الـ object نفسه

2. مثال عملي: Class للمنتجات

class Product:
    def __init__(self, name, price, quantity):
        self.name = name
        self.price = price
        self.quantity = quantity

    def get_total(self):
        return self.price * self.quantity

    def apply_discount(self, percent):
        discount = self.price * (percent / 100)
        self.price = self.price - discount
        return self.price

# نجرب
laptop = Product("Laptop", 20000, 2)
print(laptop.get_total()) # 40000

laptop.apply_discount(10) # خصم 10%
print(laptop.price) # 18000.0
print(laptop.get_total()) # 36000.0

3. الـ Inheritance = الوراثة

يعني Class يورث من Class تاني

class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        print("صوت حيوان")

class Dog(Animal): # Dog بيورث من Animal
    def speak(self):
        print(f"{self.name} بيقول هو هو 🐕")

class Cat(Animal): # Cat بيورث من Animal
    def speak(self):
        print(f"{self.name} بيقول مياو 🐈")

dog = Dog("ريكس")
cat = Cat("مشميش")

dog.speak() # ريكس بيقول هو هو 🐕
cat.speak() # مشميش بيقول مياو 🐈

4. تمرين تطبيقي: نظام بنك

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        print(f"تم إيداع {amount} جنيه")

    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            print(f"تم سحب {amount} جنيه")
        else:
            print("رصيدك غير كافي 😢")

    def show_balance(self):
        print(f"رصيد {self.owner}: {self.balance} جنيه")

# نجرب
account = BankAccount("جابر", 1000)
account.show_balance() # رصيد جابر: 1000 جنيه
account.deposit(500) # تم إيداع 500 جنيه
account.withdraw(200) # تم سحب 200 جنيه
account.show_balance() # رصيد جابر: 1300 جنيه
